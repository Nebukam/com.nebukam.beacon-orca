29 [AddComponentMenu("Nebukam/Beacon/ORCA/Obstacle Circle 2D")]
30 public class ObstacleCircle2D : ObstacleConverter<CircleCollider2D>
31 {
32
33 [Header("Circle Sampling")]
34 [Tooltip("Number of segments the circle will be converted into"), Range(3, 256)]
35 public int samples = 32;
36
37 protected override void BuildObstacles()
38 {
39
40 SetObstacleCount(1);
41 Nebukam.ORCA.Obstacle o = SetObstacleLength(0, samples);
42
43 Vector2 offset = colliderComponent.offset;
44 float3 pos = transform.position, pt = float3(0f, colliderComponent.radius, pos.z), zero = float3(0f);
45 quaternion rot = transform.rotation;
46 float inc = Maths.TAU / samples;
47
48 float3 Project(int sample)
49 {
50 float3 proj = Maths.RotateAroundPivot(pt, zero, float3(0f, 0f, inc * sample));
51 proj.x += offset.x; proj.y += offset.y; proj.z = pos.z;
52 proj = Maths.RotateAroundPivot(proj, zero, rot);
53 proj.x += pos.x; proj.y += pos.y; proj.z = pos.z;
54 return proj;
55 }
56
57 for (int i = 0; i < samples; i++)
58 o[i].pos = Project(i);
59
60 }
61
62#if UNITY_EDITOR

64 #region debug draw

66 protected override void DrawObstaclePreview(CircleCollider2D component, Color col)
67 {
68
69 float3 pos = transform.position, refloat3 = float3(0f, component.radius, 0f);
70 quaternion rot = transform.rotation;
71 float inc = Maths.TAU / samples;
72
73 float3 Project(int sample)
74 {
75 float3 proj = Maths.RotateAroundPivot(float3(refloat3.x, refloat3.y, pos.z), float3(0f), float3(0f, 0f, inc * sample));
76 proj.x += component.offset.x; proj.y += component.offset.y; proj.z = pos.z;
77 proj = Maths.RotateAroundPivot(proj, float3(0f), rot);
78 proj.x += pos.x; proj.y += pos.y; proj.z = pos.z;
79 return proj;
80 }
81
82 float3 previous = Project(samples - 1), current;
83 for (int i = 0; i < samples; i++)
84 {
85 current = Project(i);
86 DrawSegment(previous, current, col);
87 previous = current;
88 }
89
90 }
